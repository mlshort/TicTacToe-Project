270 private static Icon[] splitImg(BufferedImage img)
271 {
272 int width = img.getWidth();
273 int height = img.getHeight();
274 int subImageWidth = width / 2 - LINE_WIDTH;
275 int subImageHeight = height / 2 - LINE_WIDTH;
276 int subImageOffset = 2 * LINE_WIDTH;
277 Icon[] rgIcons = new ImageIcon[2];
278 rgIcons[0] = new ImageIcon(img.getSubimage(0, 0, subImageWidth, subImageHeight));
279 rgIcons[1] = new ImageIcon(img.getSubimage(subImageWidth + subImageOffset, 0, subImageWidth, subImageHeight));
280 return rgIcons;
281 }

324 private class GameBoardPanel extends JPanel
325 {
326 @Override
327 protected void paintComponent(Graphics g)
328 {
329 super.paintComponent(g);
330
331 if (winType != 0)
332 {
333 g.setColor(Color.RED); // Or another desired color
334 Graphics2D g2d = (Graphics2D) g;
335 g2d.setStroke(new BasicStroke(8)); // Make the line thicker
336
337 int cellSize = getWidth() / TicTacToeBoard.BOARD_SIZE; // 3;
338
339 if (winType == 1)
340 { // Horizontal win
341 g2d.drawLine(0, winIndex * cellSize + cellSize / 2, getWidth(), winIndex * cellSize + cellSize / 2);
342 boardLabels[winIndex][0].setBackground(Color.GREEN);
343 boardLabels[winIndex][1].setBackground(Color.GREEN);
344 boardLabels[winIndex][2].setBackground(Color.GREEN);
345 }
346 else if (winType == 2)
347 { // Vertical win
348 g2d.drawLine(winIndex * cellSize + cellSize / 2, 0, winIndex * cellSize + cellSize / 2, getHeight());
349 boardLabels[0][winIndex].setBackground(Color.GREEN);
350 boardLabels[1][winIndex].setBackground(Color.GREEN);
351 boardLabels[2][winIndex].setBackground(Color.GREEN);
352 }
353 else if (winType == 3)
354 { // Main diagonal win (top-left to bottom-right)
355 g2d.drawLine(0, 0, getWidth(), getHeight());
356 boardLabels[0][0].setBackground(Color.GREEN);
357 boardLabels[1][1].setBackground(Color.GREEN);
358 boardLabels[2][2].setBackground(Color.GREEN);
359 }
360 else if (winType == 4)
361 { // Anti-diagonal win (bottom-left to top-right)
362 g2d.drawLine(0, getHeight(), getWidth(), 0);
363 boardLabels[2][0].setBackground(Color.GREEN);
364 boardLabels[1][1].setBackground(Color.GREEN);
365 boardLabels[0][2].setBackground(Color.GREEN);
366 }
367 }
